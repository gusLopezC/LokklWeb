    server {
        listen 80;
        listen [::]:80;
        root /var/www/LokklWeb/dist/lokklweb/browser;
        server_tokens off;
        index index.html index.htm;

        location / {
            # First attempt to server request as file, then
            # as directory, then fall back to displaying a 404.
            try_files $uri $uri/ /index.html =404;
        }
    }
